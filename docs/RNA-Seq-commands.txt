RNA-Seq commands:

tophat

./bin/jk_compute_tophat -ref=ecoli.fa -in=t2.1.fq,t2.2.fq -out=t2_out -align_opts=-p8

./bin/jk_compute_tophat -ref=ecoli.fa -in=t3.1.fq,t3.2.fq -out=t3_out -align_opts=-p8


cufflinks

./bin/jk_compute_cufflinks -in=/t2_out/accepted_hits.bam -out=c2_out -assembly_opts=-p8 

./bin/jk_compute_cufflinks -in=/t3_out/accepted_hits.bam -out=c3_out -assembly_opts=-p8  


cuffmerge

./bin/jk_compute_cuffmerge -in=assembly.txt -ref=ecoli.fa -out=merged_out -assembly_opts=-p8 


cuffdiff

./bin/jk_compute_cuffdiff -in=/t2_out/accepted_hits.bam,/t3_out/accepted_hits.bam -out=cuffdiff_out -ref=ecoli.fa -assembly_opts=-p8 -condn_labels=C2,C3 -merged_gtf=/merged_out/merged.gtf


cuffcompare

./bin/jk_compute_cuffcompare -in=assembly.txt -out=cuffcompare_out -ref_gtf=/merged_out/merged.gtf
